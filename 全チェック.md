# ヴァルデアの遺産 全チェックスキル

## 概要
game.html・guide.html・index.htmlの全エラー・警告を一括チェックする。

## チェック項目
1. HTML基本構造（DOCTYPE・html・head・body・/html）
2. divタグの開閉対応
3. sectionタグの開閉対応
4. svgタグの開閉対応
5. CSS波括弧の対応
6. JS波括弧の対応
7. 属性クォートの整合性
8. 日本語ファイル名リンク（.htmlのみ・アンカー#は除外）
9. ノイズ文字（制御文字）
10. jumpリンク切れ（game.htmlのみ・全202件照合）
11. back-topリンク正常性（#hdrまたは#一のみ許可）
12. 必須JS関数の定義確認（applyDelta・toggleSheet・rollAll・save・load）
13. base64画像数（game.html・guide.htmlそれぞれ1件）
14. localStorage使用確認（sessionStorageが残っていないか）
15. 節数確認（113節）

## 手順
```python
import re

all_clear = True
for fname in ['game.html', 'guide.html', 'index.html']:
    with open(f'/mnt/user-data/outputs/{fname}', 'r', encoding='utf-8') as f:
        content = f.read()
    file_errors = []

    for tag in ['<!DOCTYPE html>', '<html', '</html>']:
        if tag not in content:
            file_errors.append(f"必須タグなし: {tag}")

    for tag in ['div', 'section', 'script', 'style', 'svg']:
        o = len(re.findall(f'<{tag}[\\s>]', content))
        c = content.count(f'</{tag}>')
        if o != c:
            file_errors.append(f"<{tag}>不一致: 開{o} 閉{c}")

    bad = [t for t in re.findall(r'<[^>]+>', content, re.DOTALL) if t.count('"') % 2 != 0]
    if bad:
        file_errors.append(f"属性クォート不整合: {len(bad)}件")

    for i, sb in enumerate(re.findall(r'<style[^>]*>(.*?)</style>', content, re.DOTALL)):
        if sb.count('{') != sb.count('}'):
            file_errors.append(f"CSS波括弧不一致")

    for i, sb in enumerate(re.findall(r'<script[^>]*>(.*?)</script>', content, re.DOTALL)):
        if sb.count('{') != sb.count('}'):
            file_errors.append(f"JS波括弧不一致")

    jp = re.findall(r'href="[^"#][^"]*[\u3000-\u9fff][^"]*\.html"', content)
    if jp:
        file_errors.append(f"日本語ファイル名リンク: {jp}")

    if re.search(r'[\x00-\x08\x0b\x0c\x0e-\x1f\x7f]', content):
        file_errors.append("ノイズ文字あり")

    if 'sessionStorage' in content:
        file_errors.append("sessionStorageが残っている")

    if fname == 'game.html':
        section_ids = set(re.findall(r'<div class="section" id="([^"]+)">', content))
        jump_targets = re.findall(r'href="#([^"]+)" class="jump"', content)
        missing = [t for t in jump_targets if t not in section_ids]
        if missing:
            file_errors.append(f"jumpリンク切れ: {missing[:3]}")
        backs = re.findall(r'href="([^"]+)" class="back-top"', content)
        bad_backs = [h for h in backs if h not in ['#hdr', '#一']]
        if bad_backs:
            file_errors.append(f"back-topリンク不正: {bad_backs[:3]}")
        for fn in ['applyDelta', 'toggleSheet', 'rollAll', 'function save', 'function load']:
            if fn not in content:
                file_errors.append(f"関数未定義: {fn}")
        if len(section_ids) != 113:
            file_errors.append(f"節数異常: {len(section_ids)}")

    import os
    size = os.path.getsize(f'/mnt/user-data/outputs/{fname}')
    if file_errors:
        all_clear = False
        print(f"❌ {fname}:")
        for e in file_errors:
            print(f"   {e}")
    else:
        print(f"✅ {fname}: 全項目通過 ({size:,} bytes)")

print("✅ 全クリア" if all_clear else "❌ 要修正あり")
```
